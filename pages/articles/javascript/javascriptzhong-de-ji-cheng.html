    # 原型链继承





     




    function Inherit1() {
  this.name = 'Inherit1';
}

Inherit1.prototype = new Person('Inherit1', 20); // 实现核心

const inherit1 = new Inherit1();
console.log(inherit1 instanceof Person); // true

    提示

    • 重点：让新实例的原型等于父类的实例

    • 特点：

      1.实例可继承的属性有：实例的构造函数的属性，父类构造函数属性，父类原型的属性(新实例不会继承父类实例的属性)

    • 缺点：

      • 1.新实例无法向父类构造函数传参
      • 2.继承单一
      • 3.所有新实例都会共享父类实例的属性(原型上的属性是共享的，一个实例修改了原型属性，另一个实例的原型属性也会被修改)

    # 构造函数继承


     
     





    function Inherit2(name, age) {
  Person.call(this, name, age); // 实现核心
  // Person.apply(this, arguments)
  this.age = age;
}
const inherit2 = new Inherit2('Inherit2', 30);
console.log(inherit2 instanceof Person); // false

    提示

    • 重点：用 .call() 和 .apply() 将父类构造函数引入子类函数(在子类函数中做了父类函数的自执行)

    • 特点：

      • 1.只继承了父类构造函数的属性，没有继承父类原型的属性
      • 2.解决了原型链继承缺点1,2,3
      • 3.可以继承多个构造函数属性(call多个)
      • 4.在子实例中可向父实例传参

    • 缺点：

      • 1.只能继承父类构造函数的属性
      • 2.无法实现构造函数的复用(每次用每次都要重新调用)
      • 3.每个新实例都有父类构造函数的副本，臃肿

    # 组合继承

    组合原型链继承和借用构造函数继承(常用)


     







    function Inherit3(name, age) {
  Person.call(this, name, age); // 实现核心
  this.age = age;
}
Inherit3.prototype = new Person();
Inherit3.prototype.constructor = Inherit3;

const inherit3 = new Inherit3('Inherit3', 40);

    提示

    • 重点：结合了两种模式的优点，传参和复用

    • 特点：

      • 1.可以继承父类原型上的属性，可以传参，可复用
      • 2.每个新实例引入的构造函数属性是私有的

    • 缺点：调用了两次父类构造函数（耗内存），子类的构造函数会代替原型上的那个父类构造函数

    # 寄生式组合继承

    function content(o) {
  function F() {}
  F.prototype = o;
  return new F();
}

function inheritPrototype(Child, Parent) {
  const prototype = content(Parent.prototype);
  prototype.constructor = Child;
  Child.prototype = prototype;
}

// 上面的写法可以替换为下面这种
// function inheritPrototype(Child, Parent) {
//   Child.prototype = Object.create(Parent.prototype)
//   Child.prototype.constructor = Child;
// }

function Inherit4(name, age) {
  Person.call(this, name, age);
  this.age = age;
}

inheritPrototype(Inherit4, Person);

const inherit4 = new Inherit4('inherit4', 50);

    # ES6的Class继承

    class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  sayName() {
    console.log(this.name);
  }
}

class Child extends Person {
  constructor(name, age, gender) {
    super(name, age);
    this.gender = gender;
  }
  getAge() {
    return this.age;
  }
}
