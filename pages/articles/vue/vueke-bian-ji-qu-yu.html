    # EditArea组件

    <template>
  <div
    class="edit-cell"
    :class="{
      'is-error': validateState === 'error',
      'is-validating': validateState === 'validating',
      'is-success': validateState === 'success',
    }"
    @dblclick="onFieldClick"
  >
    <div v-if="!edit && !showInput" tabindex="0" @keyup.enter="onFieldClick">
      <slot name="content" />
    </div>

    <el-tooltip
      v-if="edit || showInput"
      poper-class="table-edit-tooltip"
      :placement="tooltipPlacement"
      :open-delay="tooltipDelay"
      :disabled="disabled"
    >
      <span slot="content"> {{ validateMessage }} </span>
      <component
        :is="editableComponent"
        ref="input"
        v-model="model"
        v-bind="$attrs"
        @focus="onFieldClick"
        @keyup.enter.native="onInputExit"
        v-on="listeners"
      >
        <slot name="edit-component-slot" />
      </component>
    </el-tooltip>
  </div>
</template>

<script>
import AsyncValidator from 'async-validator'
import ElTooltip from 'element-ui/lib/tooltip'

export default {
  name: "EditArea",
  components: {
    ElTooltip
  },
  inheritAttrs: false,
  props: {
    value: {
      type: String,
      default: ''
    },
    rules: [Object, Array],
    tooltipDelay: {
      type: Number,
      default: 300
    },
    tooltipPlacement: {
      type: String,
      default: 'right'
    },
    showInput: {
      type: Boolean,
      default: false
    },
    editableComponent: {
      type: String,
      default: 'el-input'
    },
    closeEvent: {
      type: String,
      default: 'blur'
    }
  },
  data() {
    return {
      validateState: '',
      validateMessage: '',
      edit: false,
      editMode: false
    }
  },
  computed: {
    model: {
      get() {
        return this.value
      },
      set(val) {
        return this.$emit('input', val)
      }
    },
    disabled() {
      return this.validateState !== 'error'
    },
    listeners() {
      return {
        [this.closeEvent]: this.onInputExit,
        ...this.$listeners
      }
    }
  },
  watch: {
    editMode(val) {
      if (val) {
        this.edit = true
      } else {
        this.validate((message, invalidFields) => {
          if (!message) {
            this.edit = false
          }
        })
      }
    }
  },
  methods: {
    onFieldClick() {
      this.editMode = true
      this.$nextTick(() => {
        const inputRef = this.$refs.input
        if (inputRef) {
          inputRef.focus()
        }
      })
    },
    onInputExit() {
      this.editMode = false
    },
    onInputChange(val) {
      this.$emit('input', val)
    },
    validate(callback) {
      if (!this.rules || this.rules.length === 0) {
        callback()
        return true
      }
      this.validateState = 'validating'
      const descriptor = {
        value: this.rules
      }
      const validator = new AsyncValidator(descriptor)
      const model = {
        value: this.value
      }
      validator.validate(model, { firstFields: true }, (errors, invalidFields) => {
        this.validateState = !errors ? 'success' : 'error'
        this.validateMessage = errors ? errors[0].message : ''
        callback(this.validateMessage, invalidFields)
      })
    }
  }
}
</script>
